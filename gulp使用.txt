二、js合并、重命名、压缩、加密

1、js合并
(1)npm install gulp-concat --save-dev

(2)在gulpfile.js中输入代码

	var gulp = require("gulp");
	var concat = require("gulp-concat");
	gulp.task("testConcat",function(){
			gulp.src("js/*.js").pipe(concat("all.js")).pipe(gulp.dest("js/dist"));				  
	})

(3)命令提示符执行：gulp testConcat
